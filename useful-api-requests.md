## pywikibot.Page(site, 'Hovedside')
POST data:
    maxlag=5,
    format=json,
    rvprop=ids|flags|timestamp|user|comment|content,
    prop=info|revisions,
    titles=Hovedside,
    meta=userinfo,
    indexpageids=,
    action=query,
    uiprop=blockinfo|hasmsg
